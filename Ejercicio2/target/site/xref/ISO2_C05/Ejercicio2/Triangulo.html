9   public class Triangulo {

14  	private double alpha;

16  	private double delta;
17  /**
18   * Crear un triangulo
19   * @param ladoA lado 1 de un triangulo
20   * @param ladoB lado 2 de un triangulo
21   * @param ladoC lado 3 de un triangulo, en un triangulo rectangulo sería la hipotenusa
22   * @param alpha angulo entre ladoA y ladoB
23   * @param beta beta angulo entre ladoA y ladoC
24   * @param delta delta angulo entre ladoB y ladoC
25   */
26  	public Triangulo(double ladoA, double ladoB, double ladoC, double alpha, double beta, double delta) {
27  		this.ladoA = ladoA;
28  		this.ladoB = ladoB;
29  		this.ladoC = ladoC;
30  		this.alpha = alpha;
31  		this.beta = beta;
32  		this.delta = delta;
33  	}
34  	/**
35  	 * 
36  	 * @return tamaño del lado 1 de un triangulo
37  	 */
38  	public double getLadoA() {
39  		return ladoA;
40  	}
41  /**
42  * 
43  * @return tamaño del lado 2 de un triangulo
44  */

47  	}

50  *

52  *

54  	public double getLadoC() {
55  		return ladoC;
56  	}

58  *

62  		return alpha;
63  	}
64  /**
65  *

70  	}
71  
72  /**
73  * 
74  * @return angulo entre ladoB y ladoC
75  */
76  	public double getDelta() {
77  		return delta;
78  	}
79  /**
80  * 
81  * Función que devuelve el tipo de triangulo dependiendo de sus lados
82  * @param triangulo El triangulo del que se quiere conecer su tipo
83  * @return El tipo de triangulo: equilatero, isosceles o escaleno
84  */
85  	private String getTipoLadosTriangulo(Triangulo triangulo) {
86  		
87  		String tipo="";
88  		if (triangulo.getLadoA()==triangulo.getLadoB() && triangulo.getLadoB() == triangulo.getLadoC())
89  			tipo="equilatero";
90  		else if(triangulo.getLadoB()==triangulo.getLadoC()){
91  			tipo="isosceles";
92  			}
93  		else {
94  			tipo="escaleno";	
95  			
96  		}
97  
98  		return tipo;
99  	}
100 /**
101 * Función que devuelve el tipo de triangulo dependiendo de sus angulos
102 * @param triangulo El triangulo del que se quiere conecer su tipo
103 * @return El tipo de triangulo: rectangulo, obtusangulo, acutangulo
104 * @throws FalloDatos, cuando la suma de los angulos del triangulo no son iguales a 180
105 */
106 	private String getTipoAnguloTriangulo(Triangulo triangulo)throws FalloDatos{
107 		
108 		String tipo = "";
109 		double suma = triangulo.getAlpha()+ triangulo.getBeta()+triangulo.getDelta();
110 		if(suma!=180.0) {
111 			throw new FalloDatos("La suma de los angulos debe de ser 180");
112 		}
113 		if (triangulo.getAlpha()==90) {
114 			tipo = "rectangulo";
115 		}
116 		else if(triangulo.getAlpha()>90) {
117 			tipo="obtusangulo";
118 		}
119 		else{
120 			
121 			tipo="acutangulo";
122 		}
123 		return tipo;
124 	}
125 /**
126 * Funcion para obtener el tipo de triangulo
127 * @param triangulo El triangulo del que se quiere conocer su tipo
128 * @param decision obtener el tipo de triangulo. True: segun los lados, False: segun sus angulos
129 * @return tipo El tipo de triangulo
130 * @throws FalloDatos Lanzada cuando algun parametro del triangulo es menor o igual a 0 o si la suma de sus angulos en menor a 180
131 */	
132 	public String getTipoTriangulo(Triangulo triangulo,boolean decision) throws FalloDatos {
133 		String tipo="";
134 		
135 		if (triangulo.getLadoA()<=0 || triangulo.getLadoB() <=0 ||
136 			triangulo.getLadoC() <=0 || triangulo.getAlpha()<=0 ||
137 			triangulo.getBeta()<=0 || triangulo.getDelta()<=0) {
138 			throw new FalloDatos("Los lados y angulos deben ser mayor que 0");
139 		}
140 			
141 			
142 		if (decision) {
143 			tipo=getTipoLadosTriangulo(triangulo);
144 		}
145 		else {
146 			
147 			tipo=getTipoAnguloTriangulo(triangulo);
148 		}
149 
150 		return tipo;
151 	}
